Icesimba Repository

How to use

Add these orange color code to your build file (pom.xml for Maven, build.gradle for Gradle) and refresh your project.

Gradle

repositories {
  maven {
    url 'https://icesimba.github.io'
  }
}
dependencies {
  implementation '${Group}:${Name}:${Version}'
  implementation '${Group}:${Name}:${Version}@${Type}'
}

Maven

<repositories>
  <repository>
    <id>icesimba</id>
    <name>Icesimba</name>
    <url>https://icesimba.github.io</url>
  </repository>
</repositories> 
<dependencies>
  <dependency>
    <groupId>${Group}</groupId>
    <artifactId>${Name}</artifactId>
    <version>${Version}</version>
  </dependency>
  <dependency>
    <groupId>${Group}</groupId>
    <artifactId>${Name}</artifactId>
    <version>${Version}</version>
    <type>${Type}</type>
  </dependency>
</dependencies>
